import { Canvas, Meta, ColorPalette, ColorItem } from '@storybook/addon-docs'
import { config } from './configs/stitches.config'

<Meta title="Colors" />

# Colors

<ColorPalette>
  {Object.keys(config.theme.colors).map((color, index, colors) => {
    return color.startsWith('on') ? null : (
      <ColorItem
        key={index}
        title={color}
        colors={{
          [color]: config.theme.colors[color],
          [colors[index + 1]]: config.theme.colors[colors[index + 1]],
        }}
      />
    )
  })}
</ColorPalette>
